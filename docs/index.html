Документация TGBOT SOLAR API

это документация для плагина telegram bot api под solar2d, созданного @xqss_DEVELOPER. с ним можно делать крутых ботов для telegram с текстом, медиа, кнопками и прочим.

установка

кинь файл telegram_bot.lua в корень проекта и настрой бота с токеном.

local telegram = require("telegram_bot")

telegram.init({
    token = "YOUR_BOT_TOKEN",
    botName = "@YourBotName"
})

отправка сообщений

функция sendMessage шлет текст в чат.

telegram.sendMessage(chatId, "привет от solar2d!", { parseMode = "Markdown" }, function(result, error)
    if error then
        print("ошибка: " .. error)
    else
        print("сообщение ушло!")
    end
end)

отправка медиа

функция sendMedia отправляет фотки, видео, файлы или аудио.

telegram.sendMedia(chatId, "photo", "https://example.com/image.jpg", {
    caption = "пример фотки",
    parseMode = "Markdown"
}, function(result, error)
    if error then
        print("ошибка: " .. error)
    else
        print("медиа отправлено!")
    end
end)

inline-кнопки

функция sendInlineKeyboard шлет сообщение с кнопками.

telegram.sendInlineKeyboard(chatId, "выбери что-нибудь:", {
    {{text = "кнопка 1", callback_data = "btn1"}},
    {{text = "кнопка 2", callback_data = "btn2"}}
}, function(result, error)
    if error then
        print("ошибка: " .. error)
    else
        print("кнопки отправлены!")
    end
end)

обработка callback

функция answerCallbackQuery отвечает на нажатие inline-кнопок.

telegram.answerCallbackQuery(callbackQueryId, "ты нажал: " .. data, false, function(result, error)
    if error then
        print("ошибка: " .. error)
    else
        print("ответ отправлен!")
    end
end)

polling

функция startPolling запускает обработку сообщений и событий.

local function eventHandler(eventType, data)
    if eventType == "command" and data.command == "start" then
        telegram.sendMessage(data.chatId, "добро пожаловать!", nil)
    elseif eventType == "callback_query" then
        telegram.answerCallbackQuery(data.callbackQueryId, "ты кликнул: " .. data.data)
    end
end

telegram.startPolling(eventHandler)

вебхуки

функции setWebhook и deleteWebhook настраивают вебхуки для продакшена.

telegram.setWebhook("https://your-server.com/webhook", function(result, error)
    if error then
        print("ошибка: " .. error)
    else
        print("вебхук настроен!")
    end
end)

telegram.deleteWebhook(function(result, error)
    if error then
        print("ошибка: " .. error)
    else
        print("вебхук удален!")
    end
end)
